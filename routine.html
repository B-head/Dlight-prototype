RoutineDaclaration:
	(AttributeSignature,)? RoutinePrefix (,Identifier)? RoutineSignature (,InlineContext)?

RoutinePrefix:
	"rout"
	"routine"
	"func"
	"function"

RoutineSignature:
	(,TemplateSignature)? (.ArgumentSignature)? (,ReturnTypeSignature)?

ArgumentSignature:
	"(", (ArgumentList,)? ")"
	"[", (ArgumentList,)? "]"
	ArgumentList

ArgumentList:
	Argument (,",", Argument)*

Argument:
	ArgumentPart, ":=", NakedRangeLiteral
	NakedRangeLiteral, "=:", ArgumentPart
	ArgumentPart

ArgumentPart:
	(AttributeSignature,)? (VariantPrefix,)? Identifier (,":", PrefixExpression)?

ReturnTypeSignature:
	"->", "(", TupleLiteral ^ NakedRangeLitaral, ")"
	"->", "[", TupleLiteral ^ NakedRangeLitaral, "]"
	"->", TupleLiteral
